function openBibModal(bibURL, modalID) {
  const modal = document.getElementById(modalID);
  const textarea = modal.querySelector("textarea");

    .then(data => {
      textarea.value = data;
      modal.style.display = 'block';
    })
    .catch(() => {
      textarea.value = "Failed to load BibTeX.";
      modal.style.display = 'block';
    });
}
